#include <stdio.h>

int main() 
{
     int arr1[2][3]={2,3,4,5,6,7};
     int arr2[3][2]={6,7,8,9,2,3};
     
     int result[2][2];
     
     int rc =2;
     
     for(int i=0; i<2;i++)
      for(int j=0; j<2; j++)
      {
         result[i][j]=0;   
         
         for (int k=0; k<rc;k++)
         {
          result[i][j] += arr1[i][k]  * arr1[k][j];
         }
      }
      
      for(int i=0; i<2;i++)
      {
          for(int j=0; j<2; j++)
      {   
          printf("%d ",result[i][j]);
         }
         printf ("\n");
      }
    return 0;
}
