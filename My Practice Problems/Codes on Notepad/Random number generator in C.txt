Random number generating between two limits:

Code:

#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main() {

    
  int lower =25,upper =30; 
  int number;
  
  
  srand(time(NULL));
  number =(rand() %(upper - lower+1)) + lower;
  
  printf("%d\n",number);
  

    return 0;
}
